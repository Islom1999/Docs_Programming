# while tsikli
let i = 0
while (i <= 5) {
  console.log(i)
  i++
}

// 0 1 2 3 4 5

# do while tsikli
let i = 0
do {
  console.log(i)
  i++
} while (i <= 5)

// 0 1 2 3 4 5

# for of tsikli
Biz massivlar uchun for of loop dan foydalanamiz. Agar biz massivdagi har bir elementning indeksiga qiziqmasangiz, bu massivni takrorlashning juda qulay usuli.
for (const element of arr) {
  // kod shu yerda 
}


const numbers = [1, 2, 3, 4, 5]

for (const num of numbers) {
  console.log(num)
}

// 1 2 3 4 5

for (const num of numbers) {
  console.log(num * num)
}

// 1 4 9 16 25

// massivdagi barcha raqamlarni qo'shsak 
let sum = 0
for (const num of numbers) {
  sum = sum + num  // shu tarzda ham qisqartirilishi mumkin, sum += num
}
console.log(sum) // 15

# break
Break tsiklni buzish(to'xtatish) uchun ishlatiladi.
for(let i = 0; i <= 5; i++){
  if(i == 3){
    break
  }
  console.log(i)
}

// 0 1 2

Yuqoridagi kod, agar takrorlash jarayonida 3 topilsa, to'xtaydi.
# continue
Biz ma'lum bir takrorlashni o'tkazib yuborish uchun continue kalit so'zidan foydalanamiz .
for(let i = 0; i <= 5; i++){
  if(i == 3){
    continue
  }
  console.log(i)
}

// 0 1 2 4 5

# Masalalar
  1. For tsikli yordamida 0 dan 10 gacha takrorlang, while va do while yordamida xuddi shunday qiling
  2. For tsikli yordamida 10 dan 0 gacha takrorlang, while va do while yordamida xuddi shunday qiling
  3. For tsikli yordamida 0 dan n gacha takrorlang
  4. Quyidagi karra jadvalini chop etish uchun Tsikldan foydalaning:
0 x 0 = 0
1 x 1 = 1
2 x 2 = 4
3 x 3 = 9
4 x 4 = 16
5 x 5 = 25
6 x 6 = 36
7 x 7 = 49
8 x 8 = 64
9 x 9 = 81
10 x 10 = 100

  1. 0 dan 100 gacha takrorlash va faqat juft raqamlarni chop etish uchun for tsiklidan foydalaning
  2. 0 dan 100 gacha takrorlash va faqat toq raqamlarni chop etish uchun for tsikli dan foydalaning
  3. 0 dan 100 gacha takrorlash va faqat tub sonlarni chop etish uchun for tsikli dan foydalaning
  4. 0 dan 100 gacha takrorlash va barcha raqamlarning yig'indisini chop etish uchun for tsikli dan foydalaning.
